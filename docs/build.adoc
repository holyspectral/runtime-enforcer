== Build System Dependencies

On SLES/openSUSE:
[source,bash]
----
sudo zypper install -y \
  gcc \
  libelf-devel \
  clang \
  llvm \
  libbpf-devel
----

On Ubuntu/Debian:
[source,bash]
----
sudo apt-get install -y \
  build-essential \
  libelf-dev \
  clang \
  llvm \
  libbpf-dev
----

On RHEL/Fedora:
[source,bash]
----
sudo dnf install -y \
  gcc \
  elfutils-libelf-devel \
  clang \
  llvm \
  libbpf-devel
----
